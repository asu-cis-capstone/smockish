CREATE TABLE users(
user_id int unsigned auto_increment,
first_name varchar(50) NOT NULL,
last_name varchar(50) NOT NULL,
nickname varchar(20), 
city varchar(50) NOT NULL, 
state char(2) NOT NULL, 
zip char(5) NOT NULL,
referal_code int unsigned, 
email_address varchar(150) NOT NULL,
password varchar(255) NOT NULL, 
artist bit, 
collector bit, 
join_date date NOT NULL,
PRIMARY KEY(user_id), 
FOREIGN KEY(referal_code) REFERENCES users(user_id)
); 

CREATE TABLE artists(
artist_id int unsigned,
public_email varchar(150) NOT NULL, 
cv varchar(255), 
PRIMARY KEY(artist_id), 
FOREIGN KEY(artist_id) REFERENCES users(user_id)
);

CREATE TABLE artwork(
art_id int unsigned AUTO_INCREMENT,
artist_id int unsigned NOT NULL, 
title varchar(255) NOT NULL,
image varchar(255) NOT NULL, 
series varchar(255), 
year_created year(4) NOT NULL, 
medium_used varchar(100) NOT NULL, 
height varchar(4), 
width varchar(4),
price decimal(10,2) NOT NULL, 
stock int unsigned NOT NULL,  
PRIMARY KEY(art_id), 
FOREIGN KEY(artist_id) REFERENCES users(user_id)
);
